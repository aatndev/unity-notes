->To change speed of audio source or audio clip by changing the pitch
//only if you want to lower the audio while lowering the timescale
    void DoSloMo()
    {
        Time.timeScale = sloMoSpeed;
        Time.fixedDeltaTime = Time.timeScale * 0.02f;

        planeAudioSource.Stop(); //stop current audio source
        planeAudioSource.clip = slowmoAudio; //store clip
        planeAudioSource.pitch = Mathf.Clamp(Time.timeScale, 0.1f, 2f);
        planeAudioSource.Play();
    }
    
    void StopSloMo()
    {
        Time.timeScale = 1f;
        Time.fixedDeltaTime = 0.02f;
        
        planeAudioSource.Stop();
        planeAudioSource.clip = null;
        planeAudioSource.pitch = 1f;
    }
